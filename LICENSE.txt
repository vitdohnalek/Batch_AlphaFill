#Parser for command line usage
	#Requires path to a folder with targetp results with standard _'summary.targetp2' extension
	#Requires path to a folder with fasta file used for the prediction
	#Requires path to a folder where the trimmed fasta files will be written
	#Requires path/name for the .tsv file where all the information about charges will be written
	parser = argparse.ArgumentParser()
	parser.add_argument("--targetp_folder", action="store", dest="parameter_targetp_folder", type=str)
	parser.add_argument("--fasta_folder", action="store", dest="parameter_fasta_folder", type=str)
	parser.add_argument("--output_folder", action="store", dest="parameter_output_folder", type=str)
	parser.add_argument("--output", action="store", dest="parameter_output", type=str)
	args = parser.parse_args()
